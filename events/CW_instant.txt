add_namespace = exact_date_trigger

#February, 14th, 2010
country_event = {
	id = exact_date_trigger.45
	hidden = yes
	fire_only_once = yes
	
	trigger = {
		date > 2010.2.13
	}
	
	option = {
		##Maggie Thatcher Death
		if = {
			limit = { 
				country_exists = ENG
			}
			ENG = { country_event = britain.1 }
		}
		
	}
}
